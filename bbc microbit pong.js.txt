// When the BBC micro:bit runs.


function onStart(  ) {
	while (true) {
		for (var i = 0; i <= 4; i = i + 1) {
			wait(100);
			microbit.on(i, 2);

			wait(100);
			microbit.off(i - 1, 2);
		}
		for (var i = 4; i >= 0; i = i - 1) {
			wait(100);
			microbit.on(i, 2);
			wait(100);
			microbit.off(i + 1, 2);
		}
	}

}
